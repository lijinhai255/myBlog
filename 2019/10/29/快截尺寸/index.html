// 定义一个函数， 给定一个元素，返回该元素的定位值包含left值 ,top值

	function offset(dom) {
		/*
			思路:
				获取元素的定位父元素，再获取距离
				再获取元素的定位父元素的定位父元素， 再获取距离
				再获取元素的定位父元素的定位父元素的定位父元素， 再获取距离 直到 => body 让距离相累加
		 */

		 // for (; dom != document.body; dom = dom.offsetParent);


		 // 获取浏览器的信息
		 var str = window.navigator.userAgent;
		 // 定义变量简化书写
		 var isIE8 = null;
		 // 判断浏览器
		 if (str.indexOf("MSIE 8.0") === -1) {
		 	// 说明不是IE8
		 	// console.log("不是IE8")
		 	isIE8 = false;
		 } else {
		 	// 说明是IE8
		 	// console.log("是IE8");
		 	isIE8 = true;
		 }

		// 定义对象
		var result = {
			left: dom.offsetLeft,
			top: dom.offsetTop
		}


		 while (dom != document.body) {
		 	// 指向定位父元素
		 	dom = dom.offsetParent;
		 	// console.log(dom);
		 	// 判断浏览器
		 	if (isIE8) {
		 		// 说明是IE8 此时不需要加上复原素的边框
		 		result.left += dom.offsetLeft;
		 		result.top += dom.offsetTop;
		 	} else {
		 		// 不是IE8，要额外加上父元素的边框
		 		result.left += dom.offsetLeft + dom.clientLeft;
		 		result.top += dom.offsetTop + dom.clientTop;
		 	}
		}	
		// 返回result
		return result;
	}


	// 获取元素
	var box2 = document.getElementById("box2");
	// 调用
	var box2_position = offset(box2);
	console.log(box2_position);
	// jquery中获取元素到页面的方式
	console.log($("#box2").offset());

# 页面的卷动值

页面卷动值指的是：在一个页面中， 视口上面看不到的部分 （被卷起来了）
获取页面卷动值的方式：
document.documentElement.scrollTop： 新版本chrome IE 火狐 都支持
document.body.scrollTop：老版本的chrome 和 其它的一些浏览器支持

设置透明度： opacity[0-1]

# onscroll 事件

页面中的视口是一定的， 当页面的高度超过当前视口的高度， 页面中会出现滚动条，当改变滚动条位置的时候，将会触发该事件
触发该事件的条件就是改变页面卷动值
改变页面卷动值的方式有:
滚动鼠标滚轮、 键盘上的上下按键、 键盘上的空格键、 键盘上的PgUp  PgDn

	// 给document添加onscroll事件
	document.onscroll = function() {
		console.log(123);
	}

	// 触发该事件的方式有很多: 比如: 滚动鼠标滚轮、 键盘上的上下按键、 键盘上的空格键、 键盘上的PgUp  PgDn

# 截流

onscroll事件是一个高频率事件， 如果直接把代码放入事件中， 代码会被执行很多次，很浪费，为了避免代码执行多次，我们定义了一个延时器， 将要执行的代码放入延时器中， 在函数执行的时候，先清除之前延时器，最终开启新的延迟器， 从而保证代码只执行一次

// 获取元素
	var $box = $("#box");

	var timer = null;
	
	// 给document添加onscroll事件
	document.onscroll = function() {
		// console.log(123);
		// 获取页面的卷动值
		var scrollTop = document.documentElement.scrollTop;
		// console.log(scrollTop);
		// 如果页面卷动值大于或者等于2000的时候， 导航栏要出现了
		if (scrollTop >= 2000) {

			// 清除之前的延迟器
			clearTimeout(timer);

			// 开启延时器
			timer = setTimeout(function() {
				$box.animate({"opacity": 1}, 1000, function() {
					console.log("出现了");
				})
			}, 1000)
		} else {
			// 清除之前的延迟器
			clearTimeout(timer);

			// 开启延时器
			timer = setTimeout(function() {
				$box.animate({"opacity": 0}, 1000, function() {
				})
			}, 1000)
		}
	}
