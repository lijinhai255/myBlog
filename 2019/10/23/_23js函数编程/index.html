# 函数是编程

# 纯函数

  • 对于相同的输入，永远会得到相同的输出，而且没有可观察的副作用，也不会依赖外部环境的状态

# 函数的柯里化

  • 传入最少的参数 进行对参数的缓存
var checkage = min =>(age =>age>min);
var checkage18 = checkage(18)
checkage18(20) 

// 柯里化之前
function add(x,y){
    return x+y
}
add(1,2)// 3
// 柯里化之后
function addX(y){
    return function(x){
        return x+y
    }
}
addX(2)(1)//3

  • 洋葱代码h(g(f(x)))
const composes = (f, g) => (x => f(g(x)));
var first = arr=>arr[0];
var reverse = arr=>arr.reverse()
var last = composes(first,reverse);

# 高阶函数

  • 函数参数 把传入的函数做一个封装，然后返回这个封装函数，达到更高的的抽象
// 命令式
var add = function(a,b){
    return a+b
}
function math(func,array){
    return func(array[0],array[1])
}
math(add,[1,2])

# 尾调用优化

// b不是尾递归
function factorial(n){
    if(n===1)return 1;
    return n* factorial(n-1)
}
function factorial(n,total){
    if(n===1)return total;
    return factorial(n-1,n*total)
}

# 闭包

  • 虽然外层的makePowerFn函数执行完毕了，栈上的调用帧被释放，但是堆上的作用域并不被释放，因此power依旧可以呗powerFn函数访问，这样就形成了闭包
function makePowerFn(power){
    function powerFn(base){
        return Math.pow(base,power)
    }
    return powerFn;
}
var square = makePowerFn(2)
square(3)//9

# 范畴与容器

  • 函数式编程 原始目的就是求值
// 容器 Functor(函子)
var Container = function(x){
    this._value =x;
}
//函数式编程一般约定，函数有个of方法
Container.of = x=> new Container(x)
//一般约定 函子的标志就是容器有map方法，该方法将容器里面的每个值映射到另一个容器
Container.prototype.map = function(f){
    return Container.of(f(this._value))
}
Container.of(3)
    .map(x=>x+1)
    .map(x=>'Result is '+ x)

# Maybe 函子(if)

  • 由于接受各种函数，处理容器内部的值。容器内部的值可能是一个空值（比如null），而内部函数未必有处理空值的机制，如果传入控制，很可能就会出错
class Functor {
    constructor(val){
        this.val= val
    }
    map(f){
        return new Functor(f(this.val))
    }
   
}
Functor.of = function(val){
    return new Functor(val)
}
class Maybe extends Functor{
    map(f){
        return this.val? Maybe.of(f(this.val)):Maybe.of(null)
    }
}
Maybe.of(null).map(function(s){
    return s.toUpperCase()
})

var Maybe = function(x){
    this._value = x
}
Maybe.of = function(x){
    return new Maybe(x)
}
Maybe.prototype.map = function(f){
    return this.isNothing()?Maybe(null):Maybe.of(f(this._value))
}
Maybe.prototype.isNothing =function(){
    return (this._value ===null || this._value===undefined)
}

# Either

class Either {
    constructor(left,right){
        this.left = left
        this.right= right
    }
    map(f){
        return this.right?
         Either.of(this.left, f(this.right)): Either.of(f(this.left), this.right);
    }
}
Either.of = function(left,right){
    return new Either(left,right)
}
var addOne = function(x){
    return x+1
}
Either.of(5,6).map(addOne)
Either.of(1,null).map(addOne)

# AP函子

  • 函子里面包含的值，完全可能是函数
class Functor {
    constructor(val){
        this.val= val
    }
    map(f){
        return new Functor(f(this.val))
    }
   
}
Functor.of = function(val){
    return new Functor(val)
}
var addTwo = function(x){
    return x+2
}
class Ap extends Functor {   
    ap(F) {     
        return Ap.of(this.val(F.val));   
        } 
}
Ap.of = function(val){
    return new Ap(val)
}
Ap.of(addTwo).ap(Functor.of(2))
