        var d = w.document.done = false;

            try{
                d.documentElement.doScroll("left")

                setTimeout(arguments.callee,50);

            }
            init()
        })();
        d.onreadystatechange = function(){
            if(d.readyState == "complate"){
                d.onreadystatechange = null;
