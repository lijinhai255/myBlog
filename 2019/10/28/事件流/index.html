# 事件流

# call和apply

函数在定义的时候，里面的this是无法确定的， 只有当函数执行的时候才可以确定下来
简单来说： 谁调用，指向谁

call 是每一个函数天生就可以调用的方法
作用：执行函数并改变函数的this指向
使用方法：
    函数.call()
    接受多个参数：
        第一个参数就是要改变this指向
        从第二个参数开始，都是原函数所需的参数
apply与call方法的不同之处是参数的不同
    apply接受两个参数：
    第一个参数： 就是要改变的this指向
    第二个参数:  是一个数组， 数组中的每一项， 都是原函数所需要的参数

// 定义函数
function fun() {
    console.log(this, arguments);
}


// 函数自执行
// fun();

// document.onclick = fun; // 这里执行document

// document.body.onclick = fun;  // 这里指向body


// 使用call方法
// fun.call(document, 1, 2, 3, 4, 5, 6);


// 使用apply方法
fun.apply(document.body, [1, 2, 3, 4]);

# 事件流程

整个事件流程分为捕获和冒泡：
捕获： 事件从最顶层元素开始执行， 一层一层往下， 直到最精确的元素
冒泡： 事件从最精确的元素开始执行， 一层一层往上，直到最顶层的元素
当处于最精确元素的时候， 是不区分捕获和冒泡的

# DOM2级绑定事件方式

事件名称： addEventListener
这是一个方法， 每一个元素都可以调用
使用方式：
元素.addEventListener(type, fn, boolean)
type: 事件类型  注： 不带“on” 如： click, mousedown , mouseup
fn: 事件函数
boolean: 是一个布尔值，决定了事件绑定的是捕获阶段还是冒泡阶段， 默认是false, 表示冒泡阶段，如果是true， 表示绑定到了捕获阶段

# 移除事件

# DOM2级事件移除方式

事件名称：removeEventListener
作用：该方法用于移除addEventListener绑定的函数
使用方式：
dom.removeEventListener(type, fn, boolean)
type: 表示事件类型
fn: 事件函数
boolean: 是一个布尔值， 决定要移除哪个阶段的函数

// 将函数提取到外部， 并且是一个有名函数
 function click() {
	console.log("函数执行了");
}
// 绑定事件
box.addEventListener("click", click);

// 给btn添加点击事件
btn.onclick = function() {
	// 移除DOM2级事件
	box.removeEventListener("click", click);
}

# detachEvent

作用：该方法用于移除attachEvent绑定的事件
使用方式与removeEventListener一致， 只是没有了第三个参数而已

// 将函数提取到外部， 并且是一个有名函数
 function click() {
	console.log("函数执行了");
}
// 绑定事件
box.attachEvent("onclick", click);

// 给btn添加点击事件
btn.onclick = function() {
	// 移除DOM2级事件
	box.detachEvent("onclick", click);
}

# 事件绑定函数

	<div id="box"></div>

* {
    margin: 0;
    padding: 0;
}
#box {
    width: 200px;
    height: 200px;
    background-color: red;
}

	// 获取元素
	var box = document.getElementById("box");

	// 定义一个函数， 通过该函数可以实现多个浏览器的事件绑定
	/**
	 * bindEvent 实现多浏览器的事件绑定
	 * @dom 要绑定事件的元素
	 * @type 事件类型
	 * @ fn  要执行的函数
	 **/
	function bindEvent(dom, type, fn) {
		// 进行能力检测
		// 能力检测: 就是一个对象要读取一个属性， 如果能读取到就输出， 如果读取不到， 会输出undefined而不是报错的这个特点
		if (dom.addEventListener) {
			// 说明是高级浏览器
			dom.addEventListener(type, fn, false);
		} else if (dom.attachEvent) {
			// 说明是IE中的高版本
			dom.attachEvent("on" + type, fn);
		} else {
			// IE低版本或者一些不知名的浏览器
			dom["on" + type] = fn;
		}
	}
 
	// 调用
	bindEvent(box, "click", function() {
		console.log("事件执行了");
	});

# clientX clientY

这两个属性标记的是鼠标位于视口中的位置

# offsetX offsetY

这两个属性标记的是鼠标位于元素内部的位置，会受到子元素的影响

# pageX pageY

这两个属性标记的是鼠标位于页面中的位置
当一个页面打开的时候， 默认情况下是处于首屏位置， 此时鼠标位于视口中的位置和位于页面中的位置是一致的， 当页面有滚动条的时候， 改变滚动条的位置， 此时鼠标位于视口中的位置和位于页面中的位置将不一致

# screenX screenY

这两个属性标记的是鼠标位于屏幕的位置

# 拖拽

	<div id="box"></div>

* {
        margin: 0;
        padding: 0;
    }
    #box {
        /*因为是要拖拽，改变的是元素的定位值， 所以要绝对定位*/
        position: absolute;
        width: 100px;
        height: 100px;
        left: 0;
        top: 0;
        background-color: red;
    }

// 获取元素
	var box = document.getElementById("box");

	// 要求: 当鼠标按下box并拖拽的时候， 让box跟随鼠标移动

	// 添加鼠标按下事件
	box.onmousedown = function(e) {
		// 获取鼠标位于视口的位置
		var x = e.clientX;
		var y = e.clientY;

		// 获取元素的定位值
		var left = parseInt(getComputedStyle(box)["left"]);
		var top = parseInt(getComputedStyle(box)["top"]);

		// 鼠标移动事件
		document.onmousemove = function(e) {
			// 获取鼠标移动后的位置 
			var m_x = e.clientX;
			var m_y = e.clientY;
			// 元素的定位值 = 鼠标移动后的位置 - 鼠标按下的位置
			box.style.left = m_x - x + left + "px";
			box.style.top = m_y - y + top + "px";
		}
	}


	// 当鼠标抬起的时候， 取消鼠标移动事件
	document.onmouseup = function() {
		document.onmousemove = null;
	}

# 拖拽限制盒子行为

<div id="box">
		<div id="box1"></div>
</div>

* {
    margin: 0;
    padding: 0;
}
#box {
    position: relative;
    width: 800px;
    height: 400px;
    margin: 0 auto;
    border: 1px solid red;
}
#box1 {
    position: absolute;
    width: 100px;
    height: 100px;
    background-color: red;
    left: 0;
    top: 0;
}

// 获取元素
	var box = document.getElementById("box");
	var box1 = document.getElementById("box1");

	// 获取box的宽度和高度
	var box_style = getComputedStyle(box);
	var box_width = parseInt(box_style["width"]);
	var box_height = parseInt(box_style["height"]);


	// 获取box1的宽度和高度
	var box1_style = getComputedStyle(box1);
	var box1_width = parseInt(box1_style["width"]);
	var box1_height = parseInt(box1_style["height"]);

	// 给box1添加mousedown事件
	box1.onmousedown = function(e) {
		// 获取鼠标位于视口中的位置
		var x = e.clientX;
		var y = e.clientY;

		// 盒子的定位值
		var left = parseInt(getComputedStyle(box1)["left"]);
		var top = parseInt(getComputedStyle(box1)["top"]);

		// 给document添加mousemove事件
		document.onmousemove = function(e) {
			// 获取鼠标移动后的位置
			var m_x = e.clientX;
			var m_y = e.clientY;

			// 定义变量用于简化书写
			var resultX = m_x - x + left;
			var resultY = m_y - y + top;

			// 判断是否到达边界
			if (resultX < 0) {
				resultX = 0;
			} else if (resultX >= box_width - box1_width) {
				resultX = box_width - box1_width;
			}


			if (resultY < 0) {
				resultY = 0;
			} else if (resultY >= box_height - box1_height) {
				resultY = box_height - box1_height;
			}

			// 赋值box1的定位值
			box1.style.left = resultX + "px";
			box1.style.top = resultY + "px";
		}
	}


	// 当鼠标抬起的时候， 取消mousemove事件
	box1.onmouseup = function() {
		document.onmousemove = null;
	}

# 阻止冒泡

高级浏览器中的阻止冒泡方式：
1// 高级浏览器中的阻止冒泡方式
e.stopPropagation();
IE中阻止冒泡的方式：
e.cancelBubble = true

// 获取元素
	var box = document.getElementById("box");
	var box1 = document.getElementById("box1");
	var box2 = document.getElementById("box2");

	// 添加点击事件
	box.onclick = function() {
		console.log("box");
	}

	box1.onclick = function(e) {
		console.log("box1");
	}

	// 我们希望点击box2的时候， 不要再次触发box1  box的事件了
	box2.onclick = function(e) {
		console.log("box2");

		// 兼容IE8
		var e = e || window.event;

		// 高级浏览器中的阻止冒泡方式
		e.stopPropagation();


		// IE中阻止冒泡的方式， 不是一个方法而是一个属性
		e.cancelBubble = true;
	}
