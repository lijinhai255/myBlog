# 写一个轮子统一浏览器表现

	// 我们想要输出box的所有子节点， 但是childNodes属性存在兼容性问题， 在IE中和在高级浏览器中表现形式不一致
	// 所以，我们要自己想办法解决

	// 我们要定义一个函数，用来解决兼容性问题

	// 给定一个元素， 返回该元素的所有子节点并且去掉空白符

	function getChildNodes(dom) {
		// 定义一个空数组
		var arr = [];
		// 循环查看所有子节点类型， 查看是否有文本类型
		// 如果有文本类型，还要查看当前文本类型是否是纯空白
		for (var i = 0; i < dom.childNodes.length; i++) {
			console.log(dom.childNodes[i].nodeType);
			if (dom.childNodes[i].nodeType === 3) {
				// 说明是文本类型
				// 继续查看是否有空文本
				// 我们可以通过data属性来查看当前文本类型是否是纯空白
				// 定义正则表达式
				var reg = /^\s+$/;
				// 验证
				if (reg.test(dom.childNodes[i].data)) {
					console.log("第" + i + "项是一个纯空白");
				} else {
					// 说明不是空白符, 可以进入数组
					arr.push(dom.childNodes[i]);
				}
			} else {
				// 不是文本类型， 可以直接进入数组
				arr.push(dom.childNodes[i]);
			}
		}
		// 返回arr
		return arr;
	}


	// 调用
	var children = getChildNodes(box);
	console.log(children);

# 节点关系

父子

  father.childNodes[xx] // 获取某一个子节点
  father.firstChild     // 获取第一个子节点
  father.lastChild      // 获取最后一个子节点

// 父子
	// 获取ul的第一个子节点

	// 获取ul的最后一个子节点
	var last = ul.lastChild;

子父

	var father = first.parentNode;
	var father1 = last.parentNode;
