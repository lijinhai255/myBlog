# js数组精简技巧

# 1 删除数组的重复项

    var fruits = ["banner","apple","orange","apple","orange"]
    // 第一种方式
    var uniquefruits = Array.from(new Set(fruits));
    console.log(uniquefruits)
    //第二种方式
    var uniquefruits1 = [...new Set(fruits)];
    console.log(uniquefruits1)

# 2替换数组中的特定值

使用.splice(start,value to remove,valueToAdd)

    var userNames = ['小智0','小智1','小智2','小智3'];
    userNames.splice(0,2,'小智替换位置0','小智替换位置1')
    console.log(userNames)

# 3 Array.from 达到.map的效果

    var friends = [
                {name:"小智",age:22},
                {name:"张三",age:23},
                {name:"李四",age:24},
                {name:"王五",age:25},
                {name:"小六",age:21}
            ]
    var friendsName = Array.from(friends,({name})=>name)
    console.log(friendsName)

# 4 置空数组

    var friends = [
                {name:"小智",age:22},
                {name:"张三",age:23},
                {name:"李四",age:24},
                {name:"王五",age:25},
                {name:"小六",age:21}
            ]
    // var friendsName = Array.from(friends,({name})=>name)
    friends.length = 0
    console.log(friends)

# 5 将数组转为对象

    var friends = ['小智','前端小智'];
    var friendsObj = {...friends};
    console.log(friendsObj)

# 6 用数据填充数组

var newArr = new Array(10).fill('1');
console.log(newArray)//['1']

# 7 数组合并

使用展开运算符，可以将多个数组合并起来

    var friends = ['小智','前端小智'];
    var fruits = ["banner","apple","orange","apple","orange"];
    var food = [...friends,...fruits]
    console.log(food)

# 8 求两个数组的交集

1 检查数组是否重复 2 使用filter方法和includes方法

    var numOne = [1,2,3,4,5,5,6,7,6,6,8];
    var numTwo = [1,2,3,4,6,6,2,1];
    var duplicateValues = [...new Set(numOne)].filter(item=>numTwo.includes(item));
    console.log(duplicateValues)

# 9 从数组中删除虚值

在js中，虚值有false，0,"",null,NaN,undefined 使用filter()方法进行过滤虚值

    var mixedArr = [0,'blue','',NaN,9,true,undefined,'white',false]
    var trueArr = mixedArr.filter(Boolean)
    console.log(trueArr)

# 10 从数组中获取随机值

根据数组的长度获取一个随机索引

    var names = ['随机1', '随机2', '随机3', '随机4', '随机5', '随机6', '随机7', '随机8',]
    var randomName = names[Math.floor(Math.random()*(names.length))]
    console.log(randomName)

# 11 反转数组

    var names = ['随机1', '随机2', '随机3', '随机4', '随机5', '随机6', '随机7', '随机8',]
    var reverseColos = names.reverse()
    console.log(reverseColos)

# 12 lastIndexOf()方法

    var nums =[1,5,6,7,8,9,43,32,32,2,2,4,3];
    var lastIndex =nums.lastIndexOf(2)
    console.log(lastIndex)
