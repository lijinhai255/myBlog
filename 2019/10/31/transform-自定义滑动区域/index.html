# 3D

2d是平面
2d的坐标系： x + y
x表示正方向向右， y表示正方向向下

3d是立体
3d的坐标系: x + y + z
x表示正方向向右, y表示正方向向下, z垂直于屏幕

# translateZ

该属性是tansform的一个子属性，表示元素沿着自身Z轴移动了多少像素
只有该条代码，元素将不会发生任何改变
必须要给父元素添加perspective： xxpx
表示父元素距离屏幕的多少像素

    * {
			margin: 0;
			padding: 0;
		}
		#box {
			width: 560px;
			height: 300px;
			border: 1px solid red;
			margin: 0 auto;
			/*确定舞台的距离*/
			perspective: 101px;
		}
		img {
			/*演员移动的距离*/
			transform: translateZ(100px);
		}

可以是百分比
可以是单词法

    /*transform-origin: 50px 50px;*/

function initList(){
            var list = document.getElementById("list");
            var html = "";
            for(var i = 0;i < 100; i++){
                html += "<li>"+"我是"+i+"</li>";
            }
            list.innerHTML = html;
        }
       
        window.onload = function(){
            initList();
            var wrap = document.getElementById("wrap");
            var lists = document.getElementById("list");
            var startY = 0; //Y轴
            var startEl = 0;    
            var translateY = 0;
            lists.addEventListener("touchstart",(e)=>{
                startY = e.changedTouches[0].pageY;
                startEl = translateY;
            })
            lists.addEventListener("touchmove",(e)=>{
                var moveY = e.changedTouches[0].pageY;
                var distance = moveY - startY;
                console.log(distance);
                translateY = startEl +distance;
                lists.style.webkitTransform = lists.style.transform = 
                "translateY("+translateY+"px)";
            })
        }
