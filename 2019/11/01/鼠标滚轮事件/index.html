# 滚轮事件

事件名称: onmousewheel
当鼠标滚轮滚动的时候会触发该事件
该方法有兼容性的问题：
火狐不支持onmousewheel， 支持DOMMouseScroll事件

# onmousewheel

bindEvent 兼容谷歌和火狐

// 定义一个函数
function bindEvent(dom, type, fn) {
	if (type.toLowerCase() === "mousewheel") {
		// 判断浏览器信息
		var isFF = window.navigator.userAgent.indexOf("Firefox") === - 1 ? false : true;
		// console.log(isFF);
		// 如果是火狐浏览器， 那就绑定DOMMouseScroll事件
		if (isFF) {
			dom.addEventListener("DOMMouseScroll", fn, false);
			return;
		}
	}

	// 进行能力检测
	if (dom.addEventListener) {
		// 说明是高级浏览器
		dom.addEventListener(type, fn, false);
	} else if (dom.attachEvent) {console.log(1);
		// 说明是IE中的高版本
		dom.attachEvent("on" + type, fn);
	} else {console.log(2);
		// dom0级绑定方式
		dom["on" + type] = fn;
	}
}

	// 给document添加onmousewheel事件
	// document.onmousewheel = function() {
	// 	console.log("鼠标滚动了");
	// }

	// // DOMMouseScroll
	// document.addEventListener("DOMMouseScroll", function() {
	// 	console.log("火狐中滚动滚轮");
	// })


	// 统一方式
	bindEvent(document, "mousewheel", function(e) {
		// 兼容IE
		var e = e || window.event;
		console.log("事件执行了");
		// // 在高级浏览器中阻止默认事件的方式
		// if (e.preventDefault) {
		// 	e.preventDefault();
		// } else {
		// 	// IE中阻止默认事件方式
		// 	e.returnValue = false;
		// }
	})

# onmousewheel在IE中的两种绑定方式

// 给document添加onmousewheel
	// 这种绑定方式在高级浏览器中，事件对象可以被传递
	document.onmousewheel = function(e) {
		// IE中是没有将事件对象传递进来
		// 兼容IE
		var e = e || window.event;
		console.log(e);
	}

	// attachEvent这种绑定事件的方式，可以将事件对象传递进来
	// document.attachEvent("onmousewheel", function(e) {
	// 	console.log(e);
	// })
	document.addEventListener("onmousewheel", function(e) {
		console.log(e);
	})

# animate.js

function animate(dom, json, time, callBack) {

  // 计数器
  var count = 0;
  // 定义间隔时间
  var interval = 20;
  // 总次数
  var allCount= time / interval;
  
  // 计算总距离： 目标值 - 初始值
  // 定义一个对象用于保存元素的初始值
  var nowJson = {

  };
  for (var i in json) {
  	nowJson[i] = parseInt(getComputedStyle(dom)[i]);
  	// console.log(i);
  }
  // console.log(nowJson);
  // 循环完毕之后，元素的初始值也就有了


  // 得到总距离 计算步长值
  var stepJson = {

  };

  for (var i in json) {
  	// 步长值 = (目标值 - 初始值) / 总次数
  	stepJson[i] = (json[i] - nowJson[i]) / allCount;
  }

  // 循环完毕之后， 步长值就得到了
  // console.log(stepJson)

  // 赋值定时器
  var timer = setInterval(function() {
  	// 计数器++
  	count++;
  	// 不断的改变元素的定位值
  	for (var i in json) {
  		// 当前值 = 初始值 + 步长值 * 次数
  		dom.style[i] = nowJson[i] + stepJson[i] * count + "px";
  	}


  	// 判断元素是否到达边界
  	if (count >= allCount) {
  		// 停止定时器
  		clearInterval(timer);

  		// 执行回调函数
  		callBack && callBack.call(dom);
  	}
  	
  }, interval)
}

	<script type="text/javascript">
	// 获取元素
	var carousel = document.getElementById("carousel");
	var unit = document.getElementById("unit");
	// 定义宽度
	var width = carousel.clientWidth;
	// 定义length
	var length = document.getElementsByTagName("li").length - 1;

	// 定义信号量
	var idx = 0;

	// 定义锁
	var lock = true;

	// 添加鼠标滚轮事件
	bindEvent(document, "mousewheel", function(e) {
		// 节流
		if (!lock) {
			return;
		}
		// 关闭锁
		lock = false;

		if (direction(e)) {
			// 说明是往下滚动， 图片应该从右侧进入， 等价于右按钮点击事件
			// 右按钮的策略: 先拉动， 再瞬移
			idx++;
			// 拉动
			animate(unit, {"left": - width * idx}, 1000, function() {
				// 当猫腻图完全显示之后验证信号量
				if (idx > length - 1) {
					idx = 0;
					// 瞬移到真图
					unit.style.left = 0;
				}
				// 开锁
				lock = true;
			})
		} else {
			// 说明是向上滚动，图片应该从左侧进入，等价于左按钮点击事件
			// 左按钮的策略: 先瞬移， 再拉动
			idx--;
			if (idx < 0) {
				idx = length - 1;
				// 瞬移到猫腻图
				unit.style.left = -length * width + "px";
			}
			// 拉动
			animate(unit, {"left": - width * idx}, 1000, function() {
				// 开锁
				lock = true;
			})
		}
	})


	// 定义一个函数， 接收一个事件对象，返回鼠标滚轮的方向
	function direction(e) {
		// 判断e.wheelDelta是否是undefined 从而来判断是哪个浏览器
		if (typeof e.wheelDelta === "undefined") {
			// console.log("火狐");
			// 判断鼠标滚轮的方向
			if (e.detail > 0) {
				// console.log("向下滚动");
				return true;
			} else {
				// console.log("向上滚动");
				return false;
			}
		} else {
			// console.log("非火狐");
			// 判断鼠标滚轮方向
			if (e.wheelDelta > 0) {
				// console.log("向上滚动");
				return false;
			} else {
				// console.log("向下滚动");
				return true;
			}
		}
	}
	</script>

# 键盘事件

# onkeydown

当键盘被按下的时候触发该事件

	// 给box绑定onkeydown事件
	document.onkeydown = function() {
		console.log("键盘按下了");
	}

# onkeyup

当键盘被抬起的时候会触发该事件

  // onkeyup
  document.onkeyup = function() {
    console.log("键盘抬起了");
  }

# onkeypress

// onkeypress
document.onkeypress = function() {
	console.log("有字符输入了");
}

	<input type="text" id="inp" tabindex="1">
	<script type="text/javascript">
	// 获取元素
	var box = document.getElementById("box");
	var inp = document.getElementById("inp");



	// 给box绑定onkeydown事件
	box.onkeydown = function() {
		console.log("键盘按下了");
	}


	// 获取焦点
	inp.onfocus = function() {
		console.log("获取焦点");
	}

	// 失去焦点
	inp.onblur = function() {
		console.log("失去焦点");
	}
	
	/*
		总结: 如果要给document添加onkeydown事件，直接添加即可，
		 如果要给一个div元素添加onkeydown事件，
		  需要让div元素获取焦点， 
		  获取焦点的方式就是给元素添加tabIndex属性，
		  tabIndex属性值决定了按下键盘上的tab键获取焦点的顺序
	 */
	</script>
