# 存储

存储是前端仓库的意思
每一个网站分配一个
只要浏览器不卸载， 电脑不崩溃， 将永远存在
它的生命周期是永久

# 本地存储的API localStorage

存储数据：
使用方式：
    localStorage.setItem(key, value)
    key: 存储的数据名称
    value：存储的数据

获取数据：
使用方式：
    localStorage.getItem(key)
    key: 要获取的数据名称
    返回值：字符串

删除某一项数据:
使用方式：
    localStorage.removeItem(key)
    key: 要删除数据名称

删除全部：
使用方式：
    localStorage.clear()

# 会话存储 sessionStorage

除了本地存储还有一个存储对象
对象名称: sessionStorage
会话存储于本地存储的区别是生命周期
本地存储生命周期是永久
会话存储生命周期是从浏览器开启到关闭


	// 存储数据
	sessionStorage.setItem("age", 11);

	// 获取数据
	var result = sessionStorage.getItem("age");

# 多线程

js是单线程， 当js在执行时候， 页面是停止渲染的状态
    当上面函数在执行的时候（大量计算的时候）， 此时用户点击页面将不会得到任何响应（假死状态）
    
    我们希望当js在大量计算的时候， 不要阻塞当前线程
    于是H5提供了一个Worker函数， 用于开辟一个新的线程

使用方式：
1 将要大量计算的代码提取出来放到一个js文件中
2 初始化Worker函数
特点： 需要服务器环境支持

# postMessage

开辟的额外的线程用于计算， 计算的结果可以通过postMessage向主线程推送

# onmessage

额外的线程已经将信息推送到主线程中了，通过onmessage来接受信息

# WebSql数据库

openDatabase(dbname, version, decript, size)
    dbname: 数据库名称
    version: 版本号
    decript: 描述
    size: 大小
transaction
    该方法接受一个回调函数， 回调函数中的参数就是事物对象
SQLTransaction

    successCallback: 成功时候执行的回调函数
    failedCallback:  失败时候执行的回调函数

# 历史记录history

history.forward
该方法会加载历史记录列表中下一个URL
调用该方法等价于点击前进按钮或者history.go(1)

next.onclick = function() {
	// history.forward();
	// 等价方式
	history.go(1);
}

# history.back

该方法会加载历史记录列表中的前一个URL
调用该方法等价于点击了后退按钮或者history.go(-1)

// 添加点击事件
pre.onclick = function() {
    // history.back();
    // 等价方式
    history.go(-1);
}

# pushState

作用： 添加新的历史记录
使用方式：
history.pushState(obj, title, url)
    obj: 是一个对象
    title: 新的网页标题 （一般省略）
    url: 新的网页url

// 添加新的历史记录
history.pushState({"username": "老王"}, null, "index.html#aaa")
history.pushState({"username": "老王"}, null, "index.html#bbb")

# replaceState

作用： 替换当前历史记录
使用方式： 
    history.replaceState(obj, title, url)
    obj: 是一个对象
    title：新的网页标题 （一般省略）
    url: 新的网页url
replaceState与pushState区别：
    pushState就是添加新的历史记录
    replaceState是替换当前历史记录

// 添加新的历史记录
	history.pushState({"username": "老王"}, null, "index.html#aaa");
	history.pushState({"username": "老王"}, null, "index.html#bbb");
	history.replaceState({"username": "老李"}, null, "index.html#ccc");

// 获取元素
	var audio = document.getElementById("audio");
	var play = document.getElementById("play");
	var pause = document.getElementById("pause");
	var bar = document.getElementById("bar");
	var block = document.getElementById("block");
	var content = document.getElementById("content");
	var up = document.getElementById("up");
	var down = document.getElementById("down");
	var shutup = document.getElementById("shutup");

	// 添加事件
	play.onclick = function() {
		audio.play();
	}

	pause.onclick = function() {
		audio.pause();
	}


	// 给block添加mousedown事件
	block.onmousedown = function(e) {
		// 获取鼠标的位置
		var x = e.clientX;
		// 获取元素的定位值
		var left = this.offsetLeft;

		// 给document添加mousemove事件
		document.onmousemove = function(e) {
			// 获取鼠标移动后的位置
			var m_x = e.clientX;
			var result = m_x - x + left;

			// 边界判断
			if (result < 0) {
				result = 0;
			} else if (result > bar.clientWidth - block.clientWidth) {
				result = bar.clientWidth - block.clientWidth;
			}	


			// 计算进度比例
			var r = result / bar.clientWidth;
			audio.currentTime = r * audio.duration;

			// 赋值给block的定位值
			block.style.left = result + "px";
			// 改变content的宽度
			content.style.width = result + "px";
		}	
	}

	// 当鼠标抬起的时候取消mousemove事件
	document.onmouseup = function() {
		document.onmousemove = null;
	}

	// 音量+
	up.onclick = function() {
		if (audio.volume === 1) {
			return;
		}
		var volume = audio.volume * 10;
		volume++;
		audio.volume = volume / 10;
		// console.log(audio.volume);
	}
	// 音量-
	down.onclick = function() {
		if (audio.volume === 0) {
			return;
		}
		var volume = audio.volume * 10;
		volume--;
		audio.volume = volume / 10;
		// console.log(audio.volume);
	}


	shutup.onclick = function() {
		if (audio.muted) {
			shutup.innerHTML = "静音";
		} else {
			shutup.innerHTML = "恢复";
		}
		audio.muted = !audio.muted;
	}
